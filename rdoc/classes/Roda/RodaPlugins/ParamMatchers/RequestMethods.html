def match_param(key)
  if v = params[key.to_s]
    @captures << v
  end
end
match_param! (key)

Match the given parameter if present and not empty. Adds match to the captures.

def match_param!(key)
  if (v = params[key.to_s]) && !v.empty?
    @captures << v
  end
end
match_params (keys)

Match all given parameters if present, even if any/all parameters is empty. Adds all matches to the captures.

def match_params(keys)
  keys.each do |key|
    return false unless match_param(key)
  end
end
