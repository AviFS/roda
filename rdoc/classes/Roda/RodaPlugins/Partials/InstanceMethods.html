def partial(template, opts=RodaPlugins::OPTS)
  opts = parse_template_opts(template, opts)
  if opts[:template]
    template = opts[:template].split('/')
    template[-1] = "_#{template[-1]}"
    opts[:template] = template.join('/')
  end
  render_template(opts)
end
