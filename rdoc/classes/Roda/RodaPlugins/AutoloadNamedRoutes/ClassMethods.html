autoload_named_route(namespace=nil, name, file)

Autoload the given file when there is request for the named route. The given file should configure the named route specified.

40 def autoload_named_route(namespace=nil, name, file)
41   file = File.expand_path(file)
42   opts[:autoload_named_route_files] << file
43   routes = opts[:namespaced_routes][namespace] ||= {}
44   meth = routes[name] = define_roda_method(routes[name] || "named_routes_#{namespace}_#{name}", 1) do |r|
45     loc = method(routes[name]).source_location
46     require file
47     # Avoid infinite loop in case method is not overridden
48     if method(meth).source_location != loc
49       send(meth, r)
50     end
51   end
52   nil
53 end
